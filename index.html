        // 2. Fungsi Pindah Tab

        function switchTab(tabId) {

            // Sembunyikan semua tab

            document.querySelectorAll('.tab-content').forEach(el => el.classList.remove('active'));

            // Tampilkan tab yang dipilih

            document.getElementById('tab-' + tabId).classList.add('active');


            // Reset warna menu Nav Desktop

            document.querySelectorAll('.nav-btn').forEach(el => {

                el.classList.remove('text-white', 'font-bold');

                el.classList.add('text-zinc-400');

            });

            // Aktifkan warna menu Nav Desktop

            const activeDesktop = document.getElementById('nav-desktop-' + tabId);

            if(activeDesktop) {

                activeDesktop.classList.remove('text-zinc-400');

                activeDesktop.classList.add('text-white', 'font-bold');

            }


            // Reset warna menu Nav Mobile

            document.querySelectorAll('.nav-btn-mobile').forEach(el => {

                el.classList.remove('text-white');

                el.classList.add('text-zinc-500');

            });

            // Aktifkan warna menu Nav Mobile

            const activeMobile = document.getElementById('nav-mobile-' + tabId);

            if(activeMobile) {

                activeMobile.classList.remove('text-zinc-500');

                activeMobile.classList.add('text-white');

            }

        }

            let greeting = 'Selamat Malam';

            if (hour >= 5 && hour < 12) greeting = 'Selamat Pagi';

            else if (hour >= 12 && hour < 15) greeting = 'Selamat Siang';

            else if (hour >= 15 && hour < 18) greeting = 'Selamat Sore';

            document.getElementById('greeting').innerText = greeting;

        }

        updateGreeting();


        // 4. Data Tracker

        let habits = [

            { id: 1, name: 'Sholat Subuh', done: true },

            { id: 2, name: 'Sholat Dzuhur', done: true },

            { id: 3, name: 'Sholat Ashar', done: false },

            { id: 4, name: 'Sholat Maghrib', done: false },

            { id: 5, name: 'Sholat Isya', done: false },

            { id: 6, name: 'Baca Al-Quran (Minimal 1 Hal)', done: true },

            { id: 7, name: 'Sedekah / Berbuat Baik', done: false },

        ];


        function toggleHabit(id) {

            const index = habits.findIndex(h => h.id === id);

            habits[index].done = !habits[index].done;

            renderTracker();

        }


        function renderTracker() {

            const listEl = document.getElementById('tracker-list');

            listEl.innerHTML = '';

            let completed = 0;


            habits.forEach(habit => {

                if (habit.done) completed++;

                const isDoneClass = habit.done ? 'bg-emerald-900/20 border-emerald-900/50' : 'bg-zinc-800/80 border-transparent hover:bg-zinc-700';

                const textClass = habit.done ? 'text-emerald-500 line-through opacity-70' : 'text-white';

                const checkBgClass = habit.done ? 'bg-emerald-500 border-emerald-500' : 'border-zinc-500';

                const iconHtml = habit.done ? `<i data-lucide="check" class="w-3 h-3 text-white"></i>` : '';


                listEl.innerHTML += `

                    <div onclick="toggleHabit(${habit.id})" class="flex items-center justify-between p-3 md:p-4 rounded-lg cursor-pointer transition-colors border ${isDoneClass}">

                        <span class="text-sm md:text-base ${textClass}">${habit.name}</span>

                        <div class="w-5 h-5 md:w-6 md:h-6 rounded-full border-2 flex items-center justify-center ${checkBgClass}">

                            ${iconHtml}

                        </div>

                    </div>

                `;

            });


            // Update Progress UI

            const percent = Math.round((completed / habits.length) * 100);

            document.getElementById('progress-text').innerText = percent + '%';

            document.getElementById('home-tracker-status').innerText = `${completed}/${habits.length} Selesai`;

            

            // Animasi Lingkaran (menggunakan property SVG stroke-dashoffset)

            // Rumus keliling lingkaran dengan radius r=45 adalah 2 * pi * 45 ≈ 283

            const circle = document.getElementById('progress-circle');

            circle.style.strokeDasharray = "283";

            circle.style.strokeDashoffset = 283 - (283 * percent / 100);


            lucide.createIcons();

        }

        renderTracker();


        // 5. Fetch API Jadwal Sholat

        async function loadSholat() {

            try {

                // Endpoint API Publik Aladhan untuk Jakarta

                const res = await fetch('https://api.aladhan.com/v1/timingsByCity?city=Jakarta&country=Indonesia&method=11');

                const json = await res.json();

                const timings = json.data.timings;

                

                const prayers = [

                    { name: 'Subuh', time: timings.Fajr },

                    { name: 'Dzuhur', time: timings.Dhuhr },

                    { name: 'Ashar', time: timings.Asr },

                    { name: 'Maghrib', time: timings.Maghrib },

                    { name: 'Isya', time: timings.Isha }

                ];


                let html = '';

                prayers.forEach((p, idx) => {

                    // Cek aktif (contoh statis Maghrib yang aktif)

                    const isActive = p.name === 'Maghrib'; 

                    const rowClass = isActive ? 'bg-zinc-800' : 'hover:bg-zinc-800';

                    const textClass = isActive ? 'text-emerald-500' : 'text-white';

                    

                    html += `

                        <div class="flex items-center p-2 md:p-3 rounded-md transition group cursor-pointer ${rowClass}">

                            <div class="w-8 md:w-12 text-zinc-500 text-xs md:text-sm flex items-center">

                                ${isActive ? '<i data-lucide="play" class="w-4 h-4 text-emerald-500" fill="currentColor"></i>' : (idx + 1)}

                            </div>

                            <div class="flex-1 text-sm md:text-base font-medium ${textClass}">

                                ${p.name}

                            </div>

                            <div class="w-20 md:w-24 text-right flex items-center justify-end space-x-2 md:space-x-4">

                                <span class="text-xs md:text-sm ${isActive ? 'text-emerald-500 font-bold' : 'text-zinc-400'}">${p.time}</span>

                            </div>

                        </div>

                    `;

                });

                

                document.getElementById('prayer-list').innerHTML = html;

                document.getElementById('home-next-prayer').innerText = `Berikutnya: ${timings.Asr}`;

                lucide.createIcons();

            } catch (error) {

                document.getElementById('prayer-list').innerHTML = '<p class="text-zinc-500 text-sm p-4">Gagal memuat jadwal.</p>';

            }

        }

        loadSholat();
